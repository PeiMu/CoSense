74 /*
75 * Computes nCr, in a way designed to prevent overflow
76 */
77 static long
78 choose(long n, long r)
79 {
80 /*
81 * First check arguments are in correct order, and reverse if not
82 */
83 if (r > n)
84 {
85 return choose(r, n);
86 }
87
88 /*
89 * Minimize the size of intermediate results to avoid potential overflow
90 */
91 if (r < (n - r))
92 {
93 return choose(n, n - r);
94 }
95
96 /*
97 * nPr = n! / r! = (r+1)*(r+2)*...*(n-1)*n
98 */
99 long permutationsNR = 1L;
100 for (long i = r + 1L; i <= n; i++)
101 {
102 permutationsNR *= i;
103 }
104
105 /*
106 * nCr = nPr / (n-r)!
107 */
108 long nMinusRFactorial = factorial(n - r);
109 return permutationsNR / nMinusRFactorial;
110 }

113 * This is used by transformMatrixToRREF to make a swap if necessary
114 * and returns the column of the matrix at which it terminated
115 */
116 static int
117 makeRREFSwapIfNecessary(ColMajorOrderMatrixXd & matrix, int nonPivotColumnIndices[], int matrixRank, int currentRow, int currentColumn)
118 {
119 int rowCount = matrix.rows();
120 int columnCount = matrix.cols();
121
122 if (currentColumn >= columnCount)
123 {
124 return currentColumn;
125 }
126
127 int nonPivotPosition = 0;
128 while ((nonPivotColumnIndices[nonPivotPosition] != -1) && (nonPivotPosition < (columnCount - matrixRank - 1)))
129 {
130 nonPivotPosition++;
131 }
132
133 /*
134 * Only attempt to swap if current position is non-zero
135 */
136 if (matrix(currentRow, currentColumn) == 0)
137 {
138 /*
139 * Now search the matrix for a row with non-zero entry in this column
140 */
141 for (int compareRow = currentRow + 1; compareRow < rowCount; compareRow++)
142 {
143 if (matrix(compareRow,currentColumn))
144 {
145 /*
146 * Entry in compareRow is non-zero so we can make a swap
147 */
148 matrix.row(currentRow).swap(matrix.row(compareRow));
149 return currentColumn;
150 }
151 }
152
153 /*
154 * At this point we have not made a swap, try again with the next column
155 */
156 nonPivotColumnIndices[nonPivotPosition] = currentColumn;
157 return makeRREFSwapIfNecessary(matrix, nonPivotColumnIndices, matrixRank, currentRow, currentColumn+1);
158 }
159 else
160 {
161 return currentColumn;
162 }

165 /*
166 * Transforms matrix to the row reduced echelon form (RREF)
167 */
168 static void
169 transformMatrixToRREF(ColMajorOrderMatrixXd & matrix, int nonPivotColumnIndices[], int matrixRank, int currentRow = 0, int currentColumn = 0)
170 {
171 int rowCount = matrix.rows();
172 int columnCount = matrix.cols();
173
174 if (currentRow >= rowCount)
175 {
176 /*
177 * The last columns may not have been checked for swaps
178 * and should be non-pivots.
179 */
180 int columnNumber = columnCount - 1;
181 int nonPivotIndex = columnCount - matrixRank - 1;
182 while (nonPivotColumnIndices[nonPivotIndex] == -1)
183 {
184 nonPivotColumnIndices[nonPivotIndex] = columnNumber;
185 columnNumber--;
186 nonPivotIndex--;
187 }
188
189 return;
190 }
191
192 /*
193 * Make the current entry non-zero
194 */
195 currentColumn = makeRREFSwapIfNecessary(matrix, nonPivotColumnIndices, matrixRank, currentRow, currentColumn);
196 if (currentColumn >= columnCount)
197 {
198 return;
199 }
200
201 /*
202 * Divide the current row through by the current entry
203 */
204 matrix.row(currentRow) /= matrix(currentRow, currentColumn);
205
206 /*
207 * Make all other entries in current column zero by subtracting
208 * multiples of the current row (which has a value of 1 at the
209 * current position).
210 */
211 for (int q = 0; q < rowCount; q++)
212 {
213 if (q != currentRow)
214 {
215 matrix.row(q) -= matrix.row(currentRow) * matrix(q, currentColumn);
216 }
217 }
218
219 /*
220 * Repeat the algorithm, starting at the next position
221 */
222 return transformMatrixToRREF(matrix, nonPivotColumnIndices, matrixRank, currentRow + 1, currentColumn + 1);
223 }

225 static void
226 computePiGroup(ColMajorOrderMatrixXd matrix, int rowCount, int columnCount, int rank, int nonPivotColumnIndices[], ColMajorOrderMatrixXd & kernel)
227 {

229

231 {
232 /*
233 * Builds the matrix out of the non-pivot columns
234 */
235 nonPivotMatrix.col(i) = matrix.col(nonPivotColumnIndices[i]);
236 }
237
238 /*
239 * Multiply the matrix by -1
240 */
241 nonPivotMatrix *= -1;
242
243 int kernelRowCount = columnCount;
244 int identitySize = columnCount - rank;
245 ColMajorOrderMatrixXd identityMatrix = ColMajorOrderMatrixXd::Identity(identitySize, identitySize);
246
247 /*
248 * The final kernel will have as many rows as there are columns
249 * in the original matrix:
250 */
251 kernel.resize(kernelRowCount, columnCount - rank);
252
253 /*
254 * Place the identity matrix in the _rows_ corresponding to the
255 * nonpivot _columns_ in the original matrix. All other rows get
256 * the rows from the non-pivot part of the RREF matrix
257 */
258 int nextNonPivotColumnIndex = 0;
259 int nextIdentityMatrixRowToCopy = 0;
260 for (int i = 0; i < kernelRowCount; i++)
261 {
262 if ((nextNonPivotColumnIndex < (columnCount - rank)) && (i == nonPivotColumnIndices[nextNonPivotColumnIndex]))
263 {
264 kernel.row(i) = identityMatrix.row(nextNonPivotColumnIndex++);
265 }
266 else
267 {
268 kernel.row(i) = nonPivotMatrix.row(nextIdentityMatrixRowToCopy++);

275 /*
276 * For a dimensional matrix with n parameters (n columns) and rank r,
277 * (i.e., equivalently, with r pivot columns ==> r linearly-independent
278 * columns) there are choose(n, r) ways in which we can rearrange the
279 * columns of the dimensional matrix to yield a unique Pi group (see
280 * Harald Hanche-Olsen, 2004 and E. Buckingham, 1914).
281 *
282 * For each of these orderings of the r pivot columns, we generate
283 * all the possible n-bit words which have r '1's and we place the
284 * r pivot column indices at those positions.
285 *
286 * If we also wanted to account not just for placement of the pivot
287 * columns, but also for their permutations relative to each other,
288 * we could use Heap's algorithm (B. R. Heap, "Permutations by
289 * interchanges". The Computer Journal. 6 (3): 293–4) to find all the
290 * r! permutations of the r pivot column indices.
291 *
292 * Limitations:
293 * (1) We assume inputMatrix has less than 64 columns as we use a
294 * uint64_t for the pivot position bitmap.
295 */

297 static int
298 onesCount(uint64_t word)
299 {
300 int count = 0;
301 for (size_t i = 0; i < sizeof(word)*8; i++)
302 {
303 count += (word >> i) & 0x1;
304 }
305
306 return count;
307 }
